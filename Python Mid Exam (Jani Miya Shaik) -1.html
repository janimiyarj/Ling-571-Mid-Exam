def generate_bigrams(tokens):
    bigrams = []

def calculate_pmi(bigram, bigram_count, token_counts, total_bigrams):
    word1, word2 = bigram
    p_bigram = bigram_count / total_bigrams
    p_word1 = token_counts[word1] / sum(token_counts.values())
    p_word2 = token_counts[word2] / sum(token_counts.values())
    return p_bigram / (p_word1 * p_word2)
